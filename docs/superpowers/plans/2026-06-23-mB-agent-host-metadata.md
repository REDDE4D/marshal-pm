# M-B · Agent & Host Metadata Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Carry per-agent host metadata (hostname, IP, OS, arch, marshal version, host uptime) from the agent through the server to the dashboard, and surface it on the fleet view.

**Architecture:** The agent already opens the Fleet stream with a `Hello` message; extend `Hello` with host facts and extend `AgentState` (the read-path snapshot) to match. The server derives the IP from the gRPC peer (not self-reported) and stashes all of it in the in-memory `Registry`. `Registry.List()` emits the fields; the dashboard's `fleetView` serializes them; the SPA shows them on the agent header. Host uptime is carried as a static **boot time** and computed client-side.

**Tech Stack:** Go, gRPC/protobuf (`protoc` + `protoc-gen-go`), `gopsutil/v3/host`, React/TypeScript SPA.

```go
// peerIP returns the agent's remote IP from the gRPC peer (port stripped), or "".
func peerIP(ctx context.Context) string {
	p, ok := peer.FromContext(ctx)
	if !ok || p.Addr == nil {
		return ""
	}
	if host, _, err := net.SplitHostPort(p.Addr.String()); err == nil {
		return host
	}
	return p.Addr.String()
}
```

```go
// buildHello gathers static host facts for the connect handshake. Each lookup is
// best-effort: a failure leaves the field zero-valued (the dashboard shows "—").
func buildHello(name, version string) *pb.Hello {
	h := &pb.Hello{AgentName: name, MarshalVersion: version, Os: runtime.GOOS, Arch: runtime.GOARCH}
```

```tsx
function fmtUptime(bootUnix?: number): string {
  if (!bootUnix) return "";
  const s = Math.floor(Date.now() / 1000) - bootUnix;
  if (s <= 0) return "";
  const d = Math.floor(s / 86400), h = Math.floor((s % 86400) / 3600);
  return d > 0 ? `up ${d}d ${h}h` : `up ${h}h`;
}

function agentMeta(a: Agent): string {
  const parts: string[] = [];
  if (a.hostname) parts.push(a.hostname);
  if (a.os || a.arch) parts.push([a.os, a.arch].filter(Boolean).join("/"));
  if (a.marshal_version) parts.push(`marshal ${a.marshal_version}`);
  if (a.ip) parts.push(a.ip);
  const up = fmtUptime(a.host_boot_unix);
  if (up) parts.push(up);
  if (!a.connected) parts.unshift("offline");
  return parts.join(" · ");
}
```
